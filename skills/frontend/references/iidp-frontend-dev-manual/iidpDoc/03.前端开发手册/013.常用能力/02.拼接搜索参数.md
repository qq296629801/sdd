## 拼接搜索参数

- 入参： 原来的搜索过滤条件, 新加的过滤条件
- 返回： 拼接后的过滤条件
- vm.$cmd.meta.addSearch(原来的搜索过滤条件, 新加的过滤条件);

- 例如：filter = vm.$cmd.meta.addSearch(srcFilter, [['a', '=', '1']]);

```js
// 角色管理-主表格-扩展主表格接口拼接搜索参数扩展例子
base_rbac_user_menu_table_main_datasource_extend_view: {
    type: 'custom',
    selector: {
        attr: 'id',
        value: 'rbac_role_app_menu_table_main' // 主表格页面节点id
    },
    // beforeOperate: 扩展执行前的钩子函数 （见 扩展说明-扩展类型）
    beforeOperate: (app, operateItem, options) => {
        // 数据源-获取主表格
        let tableDataDsConfig = options.element.ds_config.list.find(
        (item) => item.name === 'tableData'
        );
        tableDataDsConfig.reqPrep = (vm, options, config) => {
        let srcFilter = options.params.args.filter;
        options.params.args.filter = vm.$cmd.meta.addSearch(srcFilter, [['a', '=', '1']]);
        // console.log(' === options.params.args.filter == ', options.params.args.filter);
        return options;
        };

        let tableDataCountDsConfig = options.element.ds_config.list.find(
        (item) => item.name === 'tableDataCount'
        );
        tableDataCountDsConfig.reqPrep = (vm, options, config) => {
        let srcFilter = options.params.args.filter;
        options.params.args.filter = vm.$cmd.meta.addSearch(srcFilter, [['a', '=', '1']]);
        // console.log(' === options2.params.args.filter == ', options.params.args.filter);
        return options;
        };
    }
}
```
