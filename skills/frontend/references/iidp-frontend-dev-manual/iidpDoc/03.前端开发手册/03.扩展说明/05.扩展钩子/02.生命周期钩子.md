在生命周期的钩子里返回 false 且不调用 vm.super 则逻辑中断

```js
test_hook_extend_view: {
    selector: {
      attr: 'id',
      value: 'menu_container_main'
    },
    type: 'merge',
    extend: 'xxxxxx',
    beforeOperate: (app, operateItem, options) => {
      return operateItem.view;
    },
    // 钩子
    hook: {
        // 主列表页
        gridPage: {
            created: async (vm, params, options) => {
                console.log(' === gridPage.created == params ', params);
                // 设置动态扩展 里面使用$set
                // 动态操作扩展  扩展名确保全局唯一 第一个参数是扩展名
                tech_app.setExtend(vm.$ds.idPre + '_reset_table_ops_click1', {
                    type: 'custom', // 在选中节点前面插入视图
                    selector: {
                        attr: 'id',
                        value: vm.$ds.idPre + 'table_main_table' // 关于按钮
                    },
                    beforeOperate: (app, config, options) => {
                        Tech.$set(options.element, 'bind_on_clickOptionBtn', async (params) => {
                            console.log(' 111 bind_on_clickOptionBtn ');
                            let distRes = await Super.bind_on_clickOptionBtn(params);
                            console.log(' 222 bind_on_clickOptionBtn ');
                            return distRes;
                        });
                    }
                });

                 // 如果想要主列表页不渲染 则return false 并且不调用vm.super['gridPage.created'](vm);
                await vm.super['gridPage.created'](vm);
                return false
            },
            mounted: async (vm, params, options) => {
                console.log(' === gridPage.mounted == params ', params);
                await vm.super['gridPage.mounted'](vm);
            },
            destroy: async (vm, params, options) => {
                console.log(' === gridPage.destroy == params ', params);
                await vm.super['gridPage.destroy'](vm);
            }
        },
        // 详情页
        detailPage: {
            created: async (vm, params, options) => {
                console.log(' === detailPage.created == params ', params);
                await vm.super['detailPage.created'](vm);
            },
            mounted: async (vm, params, options) => {
                console.log(' === detailPage.mounted == params ', params);
                await vm.super['detailPage.mounted'](vm);
            },
            destroy: async (vm, params, options) => {
                console.log(' === detailPage.destroy == params ', params);
                await vm.super['detailPage.destroy'](vm);
            }
        },
        // 主表格
        grid: {
            created: async (vm, params, options) => {
                console.log(' === grid.created == params ', params);
                await vm.super['grid.created'](vm);
            },
            mounted: async (vm, params, options) => {
                console.log(' === grid.mounted == params ', params);
                // await vm.super['grid.mounted'](vm);
            },
            destroy: async (vm, params, options) => {
                console.log(' === grid.destroy == params ', params);
                // await vm.super['grid.destroy'](vm);
            },
            tabs: {
                xxx: {
                    grid: {
                        created: async (vm, params, options) => {},
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    },
                    search: {
                        created: async (vm, params, options) => {},
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    },
                    drawerForm: {
                        created: async (vm, params, options) => {
                            window.ELEMENT.Message.success('打开抽屉created');
                            return await vm.super['grid.tabs.xxx.drawerForm.created'](vm);
                        },
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    },
                    addRel: {
                        created: async (vm, params, options) => {
                            window.ELEMENT.Message('添加关联created');
                            return await vm.super['grid.tabs.xxx.addRel.created'](vm);
                        },
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    }
                }
            }
        },
        // 主表格上的搜索
        search: {
            created: async (vm, params, options) => {
                console.log(' === search.created == params ', params);
                await vm.super['search.created'](vm);
            },
            mounted: async (vm, params, options) => {
                console.log(' === search.mounted == params ', params);
            },
            destroy: async (vm, params, options) => {
                console.log(' === search.destroy == params ', params);
            }
        },
        // 主表单
        form: {
            created: async (vm, params, options) => {
                console.log(' === form.created == params ', params);
                await vm.super['form.created'](vm);
            },
            mounted: async (vm, params, options) => {
                console.log(' === form.mounted == params ', params);
            },
            destroy: async (vm, params, options) => {
                console.log(' === form.destroy == params ', params);
            },
            tabs: {
                xxx: {
                    grid: {
                        created: async (vm, params, options) => {},
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    },
                    search: {
                        created: async (vm, params, options) => {},
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    },
                    drawerForm: {
                        created: async (vm, params, options) => {
                            window.ELEMENT.Message.success('打开抽屉created');
                            return await vm.super['form.tabs.xxx.drawerForm.created'](vm);
                        },
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    },
                    addRel: {
                        created: async (vm, params, options) => {
                            window.ELEMENT.Message('添加关联created');
                            return await vm.super['form.tabs.xxx.addRel.created'](vm);
                        },
                        mounted: async (vm, params, options) => {},
                        destroy: async (vm, params, options) => {},
                    }
                }
            }
        },
        // 树
        tree: {
            created: async (vm, params, options) => {
                console.log(' === tree.created == params ', params);
                await vm.super['tree.created'](vm);
            },
            mounted: async (vm, params, options) => {
                console.log(' === tree.mounted == params ', params);
            },
            destroy: async (vm, params, options) => {

        }
    }
```
