## 表单查询

工具栏新增按钮是否可点击

```js
hook: {
  form: {
        // 表单数据查询前
        beforeQuery: async (vm, params, options) => {
            // params: 表单查询参数
          console.log(vm, params, options, '==beforeQuery==');
          params.aaa = 1;
          return params; // 需要返回接口参数
        },
        // 表单数据查询
        query: async (vm, params, options) => {
            // params: 表单查询参数
          console.log(vm, params, options, '==query==');
          return await vm.super['form.query'](vm, params, options); // 返回查询到的表单数据
        },
        // 表单数据查询后
        afterQuery: async (vm, params, options) => {
            // params：表单数据
          console.log(vm, params, options, '==afterQuery==');
          window.ELEMENT.Message.success('表单数据查到啦');
          return await vm.super['form.afterQuery'](vm, params, options);
        }
  }
}
```

## 表单初始化

在表单查询后初始化

```js
hook: {
  form: {
    // 表单为编辑情况时，params为表单数据
    // 表单为新增情况时，params为空对象
    init: async (vm, params, options) => {
      console.log(options.formMain, "==主表单节点==");
      if (options.isEdit) {
        console.log("====编辑====");
        params.display_name = 1;
        options.formMain.items = [];
      } else if (options.isCreate) {
        console.log("====新增====");
        params.display_name = 2;
        options.formMain.items = [
          {
            type: "text",
            value: "test",
          },
        ];
      }
      return params;
    };
  }
}
```

## 表单保存

```js
hook: {
  form: {
        // 保存表单时的校验
        validate: async (vm, params, options) => {
            // params: 表单数据
          return await vm.super['form.validate'](vm, params, options);
        },
        // 点击保存按钮保存前 options.triggerType参数用来区分新增和编辑
        beforeSave: async (vm, params, options) => {
            // params: 表单保存接口参数
          console.log(vm, params, options, '==beforeSave==');
          params.test = 'test';
          return params;
        },
        // 保存 options.triggerType参数用来区分新增和编辑
        save: async (vm, params, options) => {
            // params: 表单保存接口参数
          console.log(vm, params, options, '==form save==');
          return await vm.super['form.save'](vm, params, options);
        },
        // 保存后 options.triggerType参数用来区分新增和编辑
        afterSave: async (vm, params, options) => {
            // params: 保存接口返回的数据
          console.log(vm, params, options, '==afterSave==');
          return await vm.super['form.afterSave'](vm, params, options);
        }
  }
}
```
