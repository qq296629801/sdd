## 是否可新增

新增按钮（节点后的新增、工具栏上的新增）是否可点击

```js
hook: {
  tree: {
    canCreate: async (vm, params, options) => {
      // triggerType: treeNodeCreate 节点后的新增 / tBarCreateBtn 工具栏上的新增
      if (options.triggerType === "treeNodeCreate") {
        // triggerType=treeNodeCreate时params为当前节点数据
        return true;
      }
      return false;
    };
  }
}
```

## 保存

新增或编辑时的保存，包含弹窗表单及右侧区域表单

```js
hook: {
  tree: {
        beforeSave: async (vm, params, options) => {
          console.log(options.triggerType) // rightFormSave:右侧表单保存 / dialogFormSave:弹窗表单保存
          params.other = 1;
          return params;
        },
        save: async (vm, params, options) => {
          return await vm.super['tree.save'](vm, params, options);
        },
        afterSave: async (vm, params, options) => {
          return await vm.super['tree.afterSave'](vm, params, options);
        },
  }
}
```

## 删除

```js
hook: {
  tree: {
    beforeDelete: async (vm, params, options) => {
        // params: 删除接口参数
        return params;
    },
    delete: async (vm, params, options) => {
        return await vm.super['tree.delete'](vm, params, options);
    },
    afterDelete: async (vm, params, options) => {
        return await vm.super['tree.afterDelete'](vm, params, options);
    },
  }
}
```

## 查询树数据

```js
hook: {
  tree: {
    beforeQuery: async (vm, params, options) => {
        console.log(params) // 查树数据接口参数
        return params; // 需要返回调接口的参数
    },
    query: async (vm, params, options) => {
        console.log(params) // 查树数据接口参数
        // 调用上一层继承的方法 需要返回接口返回的数据
        return await vm.super['tree.query'](vm, params, options);
    },
    afterQuery: async (vm, params, options) => {
        console.log(params) // 树数据
        return params;
    }
  }
}
```

## 选中节点、取消选中

```js
hook: {
  tree: {
    select: async (vm, params, options) => {
        console.log(params) // 有checkbox时params为勾选的数据 没有checkbox时params为当前点击的数据
        console.log(options.triggerType) // checkbox / treeNodeClick
        console.log('===triggerType=checkbox时===', options.checked) // 已勾选的数据
        return params;
    },
    cancelSelect: async (vm, params, options) => {
        console.log(params) // 当前取消勾选的数据
        console.log(options.triggerType) // checkbox
        console.log(options.checked) // 已勾选的数据
        return params;
    }
  }
}
```
