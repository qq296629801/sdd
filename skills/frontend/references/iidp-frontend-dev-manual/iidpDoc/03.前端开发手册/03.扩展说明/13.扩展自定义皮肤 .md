**2. 右上角皮肤下拉框增加新选项**

- 引入全局变量文件 green_skin.js

```js
import green_skin from "./skins/green_skin";
```

- 前端拓展增加主题皮肤

::: details 查看代码

```js
custom_theme_skin_extend: {
  type: 'custom',
  selector: {
    attr: 'id',
    value: 'template-meta-page-app'
  },
  beforeOperate: (app, operate, options) => {
    const themesList = [
      {
        value: 'CUSTOM-GREEN', // 皮肤标识
        label: '黄瓜绿', // 皮肤名
        color: '#779977', // 皮肤颜色
        options: {
          // 皮肤在下拉项的 class 样式 - 类名和样式都需要同步修改
          className: 'button_meta_header_skin_green',
          css: `
              .button_meta_header_skin_green .meta-el-icon-point-green {
                display: inline-block;
                width: 0.07rem;
                height: 0.07rem;
                margin-right: 0;
                border-radius: 50%;
                background-color: #779977;
              }
          `,
          options: {
            icon: 'meta-el-icon-point-green'
          }
        }
      }
    ];

    //将平台原本皮肤与自定义皮肤进行合并
    options.element.theme.list = [...options.element.theme.list, ...themesList];

    //定义全局样式
    const themeClass = {
      'CUSTOM-GREEN': green_skin
    };

    //设置默认皮肤颜色
    const themeStore = JSON.parse(localStorage.getItem('TECH-THEME'));
    if (!themeStore || (themeStore && !themeClass[themeStore.value])) {
      localStorage.setItem('TECH-THEME', JSON.stringify(options.element.theme.list[0]));
    }

    options.element.theme.class = {
      ...options.element.theme.class,
      ...themeClass
    };
  }
},
```

import greenSkinCss from "./styles/greenSkinCss";

- 前端拓展监听主题色，动态增加样式

::: details 查看代码

```js
demo_watch_hearder_theme_data_change_extend_view: {
  type: 'custom',
  selector: {
    attr: 'id',
    value: 'dropdown_meta_header_skin'
  },
  beforeOperate: (app, operateItem, options) => {
    options.element.created = (vm) => {
      let skinNode = tech_app.page.getNode('dropdown_meta_header_skin');
      let skinDataReactive = window.Tech.reactiveFn(skinNode?.$ds);
      if (skinNode.data._demo_skin_Watch) {
        // 检查之前是否有对应watch 有则先销毁
        skinNode.data._demo_skin_Watch();
        skinNode.data._demo_skin_Watch = null;
      }
      // 创建watch
      skinNode.data._demo_skin_Watch = window.Tech.watchFn(
        () => skinDataReactive.skinText,
        (val, old) => {
          const themeCacheKey = window.Tech?.theme?.cacheKey || 'TECH-THEME';
          // 获取当前主题
          const themeDefine = JSON.parse(localStorage.getItem(themeCacheKey));
          // 执行需要动态触发的逻辑 - 此案例为动态增加样式
          if (themeDefine.value == 'CUSTOM-GREEN') {
            // 动态增加样式 - addClass
            tech_app.utils.addClass('green_theme_css', greenSkinCss );
          } else {
            tech_app.utils.removeClass('green_theme_css');
          }
        },
        { deep: true, immediate: true }
      );
    };
  }
}
```
