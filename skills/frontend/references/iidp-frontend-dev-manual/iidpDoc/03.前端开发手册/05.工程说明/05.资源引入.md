# 资源引入

前端依赖的第三方库，或组件库的资源地址引入

## 资产包外部资源
./apps/[自己的扩展应用]/common/assets.json：
```json
{
    "lib": [ // 前端依赖的第三方库
      {
        "title": "jq", // 第三方库标题 确保唯一
        "version": "1.9.1", // 第三方库版本号
        "urls": [ // 第三方库连接路径
            "https://cdnmanhattan.mshengedu.com/cdnFile/lib/jquery.min.js"
        ]
      }
    ],
    "packages": [ // 组件库
      {
        "title": "web-lib 组件库", // 组件库名字
        "package": "@tech/web-lib", // 组件库包名
        "version": "0.0.15", // 组件库版本号
        "export": "web-lib", // 导出的包名
        "urls": [ // 组件库连接 运行时
          "http://192.168.175.190:4877/@tech/web-lib@0.0.17/style.css",
          "http://192.168.175.190:4877/@tech/web-lib@0.0.17/build.web-lib.umd.js"
        ],
        "dependencies": [ // 组件库需求依赖的第三方库 (可选项)
          {
            "lib": "elementUi",
            "version": "1.0.0"
          }
        ]
      }
    ]
  }
```

## 资产包内部资源
./apps/[自己的扩展应用]/common/assetsImport.js：
```js
// 扩展ui组件库，需要基于element开发的组件库
import MYUI from '../../../../xxx/my-ui-index.js';
export default {
  'my-ui' : { // 引入自定义组件库
    isComponent: true,
    comp: MYUI
  }
};
```

```js
// 引入已有的库
import moment from 'moment';
window.moment = moment; // 将库赋值到全局

import myCommonJs from '../xxx/xxx/myCommonJs.js'
window.myCommonJs = myCommonJs //可以自定义一个全局属性放入自定义的js
export default {
  // 'my-ui' : { // 引入自定义组件库
  //   isComponent: true,
  //   comp: MYUI
  // }
};
```

对于有引入顺序要求的引入步骤：

1、先创建一个引入的js,假设创建`app\assets\importJQ.js`文件
```js
import $ from './jquery.mini.js';
//import $ from 'jquery';

window.$ = $;
// window.myLib.$ = $ //可以放到自定义的全局变量
window.jQuery = $;
// window.myLib.jQuery = $ //可以放到自定义的全局变量

```

2、在`app\common\assetImport.js`文件中引入创建好的js
```js

import '../assets/importJQ.js';
import 'xxxx.js' //引入依赖jquery的文件

export default {
  // 'my-ui' : { // 引入自定义组件库
  //   isComponent: true,
  //   comp: MYUI
  // }
};
```
