# 配置iframe性能模式
> 工程 config/apps.json 添加 TABIFRAME配置： host：iframe父级域名，contentHost：子iframe域名

> 注意配置的域名或ip，以http开头，结尾不需要斜杆 /

> 配置的两个域名或ip 必须一级域名不同，或ip不同， 否则起不到新建渲染进程提高性能的效果
```json
{
  "self": {
    "master": {
      "demo": "/umdComps/tech-demo/config/app.json"
    }
  },
  "apps": {
    "demo": {
      "master": {
        "tech-demo": "/umdComps/tech-demo/config/app.json"
      }
    }
  },
  "templateApp": "TechMetaPage",
  "global": {
    "master": {
      "apiHost": "/api"
    },
    "TABIFRAME": {
      "host": "http://localhost.com",
      "contentHost": "http://content.test.com"
    },
    "routerBase": "/iidp/"
  }
}
```

> js判断是否开启TABIFRAME模式：
```js
if (tech?.TABIFRAME) {
	// 开启了iframe模式
} else {
	// 没有开启了iframe模式
}
```

> js判断是否在子iframe里面：
```js
if (tech?.TABIFRAME && !tech?.TABIFRAME?.inner) {
	// 在iframe外层
}
if (tech?.TABIFRAME?.inner) {
	// 在子iframe里面
}
```

> iframe间通讯公共方法 pageFn：
```js
// 在子iframe 通知父级运行 打印console.log
if (tech?.TABIFRAME?.inner) {
	// 第一参数 iframe选取标识，parent则选定父iframe, 其他字符串为子iframe的id
	// 第二参数 是传给第三个参数的 params
	// 第三参数 在对应选定iframe上运行的回调方法
	Tech.pageFn('parent', {a:1}, (params) => {
		console.log(params.a) // 在父级打印出 1
	})
}

// 在父级 通知子iframe运行 打印console.log
if (tech?.TABIFRAME && !tech?.TABIFRAME?.inner) {
	Tech.pageFn('子iframe的id', {a:1}, (params) => {
		console.log(params.a) // 在子iframe打印出 1
	})
}
```
