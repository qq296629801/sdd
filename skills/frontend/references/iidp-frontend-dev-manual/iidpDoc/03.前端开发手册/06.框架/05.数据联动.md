# 数据联动
## 省市联动
```js
{
	type: 'form',
	dataSource: {
		form: {
			province: '',
			city: ''
		}
	},
	items: [
		{
			type: 'select',
			text: '省',
			name: 'province',
			ds_config: {
			  	type: 'meta',
				name: 'data1',
				autoRequest: true, // 在节点渲染后，是否自动触发请求，默认false
				options: {
					params: { // 参考元模型api传参规范
						args: {
							properties: ['name']
						},
						service: 'search',
						model: 'xxx'
					}
				}
			},
			bind_options: '$ds.data1'
		},
		{
			type: 'select',
			text: '市',
			name: 'city',
			ds_config: {
			  	type: 'meta',
				name: 'data2',
				doWhenChange: 'reload',  // 绑定值发生改变后重新加载
				options: {
					params: { // 参考元模型api传参规范
						args: {
							bind_filter: '${[["id", "=", $ds.form.province]]}',   // 绑定表单的省份
							properties: ['name']
						},
						service: 'search',
						model: 'xxx'
					}
				}
			},
			bind_options: '$ds.data2',
		}
	]
}
```

## 显隐联动
```js
{
	type: 'text',
	bind_display: '${$ds.form.b === 1}'
}
```
