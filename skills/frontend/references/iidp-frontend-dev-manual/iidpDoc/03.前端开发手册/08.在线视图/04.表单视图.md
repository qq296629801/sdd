# 表单视图

- columns：表单项
- tabs： 子表项
- tbar： 工具栏按钮 （t-base 2.4 或以上版本可以使用）

## 原始后端视图

```js
{
  "type": "form",
  // 表单项
  "columns": ["name", "login", "mobile", "email", "role_ids", "password", "status"],
  "tabs": [
    {
      // 子表tab标题
      "header": "角色",
      // 子表tbar按钮
      "tbar": [
        {
          "name": "添加",
          "action": "addEr",
          "auth": "update"
        },
        {
          "name": "删除",
          "action": "deleteEr",
          "auth": "delete"
        }
      ],
      // 子表配置
      "body": {
        "type": "grid,form,search",
        "field": "role_ids",
        "columns": ["name", "is_admin"]
      }
    }
  ]
}
```

更多配置请看[后端-基本视图-form](/pages/405eb0/#_3-8-1-1-表单-form)

## 控制表单项的显示隐藏

动态控制显示隐藏

```js
"columns": [
  "name",
  "status",
  {
    "name": "prop1",
    "bind_display": "${$ds.form.status === 1}"   // 当status 等于 1 时显示
  },
  {
    "name": "prop2",
    "bind_display": "${$ds.form.status === 2}"   // 当status 等于 2 时显示
  },
]
```

同一个表单项不同展现形式, 新增或编辑情况下控制同一个字段 不同配置

- 定义不同的 id

```js
// 该表单项在新增是必填，展示下拉控件；编辑是非必填，展示开关控件
"columns": [
    {
        "displayName": "作者1",
        "id": "book_form_main_top_common_items_0_items_login1",
        "name": "author",
        "custom": true,
        "required": false,
        "type": "switch",
        "bind_display": "${$ds.clickType == 'edit'}"

    {
        "displayName": "作者2",

        "name": "author",
        "custom": true,
        "required": true,
        "type": "select",
        "bind_display": "${$ds.clickType == 'add'}"
    }
]
```

## 自定义表单项属性

```js
{
  "type": "form",
  "columns": [
    {
      "name": "name",
      "displayName": "角色名称",
      "custom": true
    },
    "status",
    {
      "name": "prop1",
      "required":true
      "custom": true
    },
  ]
}

配置说明：
  custom: true  默认false,配置为true时，视图配置的属性覆盖fields里对应字段定义的属性

注：
修改表单项名称，可配置displayName或text,建议用displayName
```

## 自定义表单项值

```js
{
  "type": "form",
  "transformInitValue": (form)=>{form.code = form.code + '111'; return form},
  "columns": [
    "name",
    "code",
    "status",
  ]
}

配置说明：
  transformInitValue: 对表单数据的处理
  入参form为当前表单数据

注：
 重新赋值后必须retrun form
```

## 修改表单项校验规则

配置 rules

```js
"columns": [
  "name",
  "status",
  {
    "name": "mobile",
    "rules": [
      {
        "required": true,
        "trigger": 'blur',
        "message": '请输入企业名称'
      },
      {
        "validator": (rule, value) => {
          if (rule.value === '') {
            rule.callback(new Error('请输入手机号码'));
          } else if (!/^1[3|4|5|6|7|8|9][0-9]\d{8}$/.test(rule.value)) {
            rule.callback(new Error('请输入正确的手机号码'));
          } else {
            rule.callback();
          }
        },
        "trigger": 'blur'
      }
    ]
  },
]
```

```js
{
```

```js
    "path": "$ds.clickType",
    "transform": (params) => {
      return params.self.$ds.clickType != 'add';
```

```js

  "name": "app_ids",
```

## 表单项列宽配置

通过 span 属性配置表单项列宽【最佳实践】，搭配 row 属性控制同行排列 

规则：一行总宽为 24，按比例填写 span 值

![](./img/BOOK_09.jpg)

```js
{
  "type": "form",
  "columns": [
    {
      "name": "fieldName",
      "row": 1,        // 相同 row 值的字段排在同一行
      "span": 6,       // 占 24 分之 6，即 1/4 宽
      "minWidth": 300  // 最小宽度（可选）
    }
  ]
}
```

### 3 tabs 可编辑表格-子表 er 关系表配置删除不调接口，只做可视删除

![](./img/isAloneSave.png)

```js
"tabs": [
  {
    "header": "用户",
    "isUiDelete": true, // 删除是否仅可视删除，不调用删除接口，仅支持在下表中使用
    "body": {
      "gridEditMode":{
        "mode": "multipleRow", // multipleRow:多行编辑 singleRow:单行编辑
        // 行内编辑需要用到的类型 如果只修改不新增删除就只写update
        // 存在update： 工具栏会出现一个编辑按钮
        // 存在add且工具栏存在新增按钮： 会把新增按钮替换成新增行按钮
        // 存在delete：点击编辑按钮操作列内会出现删除按钮
        "rowBtns": ["update", "add", "delete"],
        "defaultCancelEdit": true, // 配置上下表格的子表视图,默认进入编辑模式
        "addRowAutoFocus": false // 行内编辑新增行时是否自动获取焦点，默认为true (2.4以上uat版本)
      },
    }
  },
],
```

### 7 tabs 动态子表

- hideRelForm: (vm, val)=>{ return true } 隐藏 tabs

- beforeEditMethod: ({ row, rowIndex, column, columnIndex })=>{ return false } 行内编辑动态控制列不可编辑

注：2.4 以上版本

```js
{
  "type": "form"，
  "columns": [
    "itemCode",
    "itemName",
    "unitId"
  ],
  "tabs": [
    {
      "header": "物料属性列表",
      "rowspan": 3,
      "body": {
       "type": "grid,form",
        "model": "example_item_attribute",
        "field": "itemAttributeList",
        // 动态隐藏子表 val是表单数据
        "hideRelForm": (vm, val) => {if (val.itemCode == '123') return true;},
        "gridEditMode": {
          "mode": "multipleRow",
          "rowBtns": [
            "update",
            "add",
            "delete"
          ],
          "defaultCancelEdit": true
        },
        // 行内编辑动态控制可编辑列
        "beforeEditMethod": ({ row, rowIndex, column, columnIndex }) => {if (row.color == 123 && column.field == 'price') { return false;}return true;},
        "columns": [
          {
            "name": "color",
            "rowEditable": true
          },
          {
            "name": "price",
            "rowEditable": true
          },
          {
             "name": "weight",
             // 动态隐藏子表列 val是表单数据
             "hideRelForm": (vm, val) => { if(val.itemCode == '1234') return true},
          }
          "itemAttributeType"
        ]
      }
    }]
},
```

## tab 配置子表或者孙表是否根据条件查询

- tabs 配置"controlTableSearchFn" 方法， return true 表示允许查询，false 表示不查询
-  (vm, row) => 出参： vm当前子表或者孙表id， row 父级当前行数据
- 默认情况下，在 `return false (不查询)` 时，会清空表格数据，配置 `disabledSubTableClearOnSearch: true` 可跳过表格清空逻辑

```js
{
  "type": "form",
  "tabs": [
    {
      "header": "用户",
      "controlTableSearchFn": (vm, row) => { if (row.id == 'xxxx'){return false} else {return true} },
      "disabledSubTableClearOnSearch": true,
      "body": {
        "type": "grid,search",
        "field": "user_ids",
        "columns": [
          "login",
          "name"
        ]
      }
    }
  ]
}
```

## 表单项 label 内嵌进控件

```js
    }

  ]
}
// apps.json
{
  "self": {xxx},
  "apps": {xxx},
  "templateApp": "TechMetaPage",
  "global": {xxx},
  "innerLabel": true // 为true则表单项内嵌label
}
```
